Q) Table of 2 using functions(for, while and do-while)

Sol:
#include <stdio.h>

int for1()
{

    int i,s=2,out;

    printf("For : \n");

    for(i=1;i<4;i++)

    {

        out=s*i;

        printf("2 * %d = %d\n",i,out);

    }

}

int while1()
{

   int i=1,s=2,out;

   printf("While : \n");

   while(i<4)

   {

     out=s*i;

        printf("2 * %d = %d\n",i,out);

        i++;

   }

}

int dowhile1()
{

   int i=1,s=2,out;

   printf("Do While : \n");

   do{

       out=s*i;

        printf("2 * %d = %d\n",i,out);

        i++;

   }

   while(i<4);

}

main()
{

    for1();

    while1();

}

5)Write a C program to print all Armstrong numbers between 1 to n.

Sol:
#include <stdio.h>
#include <math.h>

int main()
{
    int num, lastDigit, digits, sum, i, end;

    
    printf("Enter upper limit: ");
    scanf("%d", &end);

    printf("Armstrong number between 1 to %d are: \n", end);

    for(i=1; i<=end; i++)
    {
        sum = 0;

      
        num = i;

        
        digits = (int) log10(num) + 1;

        
        while(num > 0)
        {
            
            lastDigit = num % 10;

           
            sum = sum + pow(lastDigit, digits);

            
            num = num / 10;
        }

      
        if(i == sum)
        {
            printf("%d, ", i);
        }

    }

    return 0;
}
